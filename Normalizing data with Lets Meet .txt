 1. Executed DESCRIBE and SELECT statement on the grp_member table
 2. Reviewed the PRIMARY KEY on the Grp_member table and determined no Primary key had been set up yet.

-- To ensure grp_member table complies with 3NF, the following changes were implemented:
-- 1. Created a table to separate the information about members and the information about groups. 


CREATE TABLE group_sign_ups 
AS SELECT DISTINCT
        member_id,
        group_id,
        joined
FROM grp_member;


CREATE TABLE members 
AS SELECT DISTINCT 
        member_id,
    member_name,
    member_status,
    hometown,
    city
FROM grp_member;


-- 2. Added Primary and foreign keys to connect both tables.
ALTER TABLE members
ADD PRIMARY KEY (member_id);


ALTER TABLE group_sign_ups
ADD FOREIGN KEY (member_id)
REFERENCES members(member_id);


ALTER TABLE group_sign_ups
ADD FOREIGN KEY (group_id)
REFERENCES grp (group_id);


SELECT 
    *
FROM members;
DESCRIBE members;


SELECT 
    *
FROM group_sign_ups;
DESCRIBE group_sign_ups;


SELECT 
    g.member_id,
    g.group_id,
    g.joined,
    member_name,
    member_status,
    hometown,
    city
FROM
    members AS m
        JOIN group_sign_ups AS g ON m.member_id = g.member_id;
